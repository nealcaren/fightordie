# Function to recursively gather .qmd files from a directory and its subdirectories
def gather_qmd_files(directory):
    qmd_files = []
    for root, _, files in os.walk(directory):
        for file in files:
            if file.endswith(".qmd"):
                qmd_files.append(os.path.join(root, file))
    return qmd_files


# Function to load documents from .qmd files
def load_documents(directory):
    files = gather_qmd_files(directory)
    documents = {}
    for file in files:
        with open(file, "r") as f:
            documents[file] = f.read()
    return documents

def make_cite(md_text):
    for line in md_text.splitlines():
        if line[:7] == "title: ":
            title = line.replace("title: ", "")

        if line[:6] == "date: ":
            date = line.replace("date: ", "")
            year = date.split(" ")[-1]

    if title[0] == "'" and title[-1] == "'":
        title = title[1:-1]

    cite = f"[{title} ({year})]"
    return cite


def make_link(fn):
    return fn.split("fightordie/")[-1]


def make_reference(fn,result_dict):
    cite = make_cite(result_dict[fn])
    link = make_link(fn).replace(".qmd", ".html")
    reference = f"{cite}(/{link})"
    return reference

def make_similar_appendix(article_text):
    results = collection.query(
        query_texts=[f'{article_text}'],
        n_results=6
    )
    # print(results['distances'])
    result_dict = {key: value for key, value in zip(results['ids'][0][1:], results['documents'][0][1:])}
    similar_list = [make_reference(fn ,result_dict) for fn in result_dict.keys()]
    similar = '* '+ '\n* '.join(similar_list)
    prefix = '\n<!-- Similar articles start here -->\n\n<hr>\n#### Related Articles:\n\n'
    suffix = '\n\n<!-- Similar articles end here -->'
    similar = prefix + similar + suffix
    return similar
